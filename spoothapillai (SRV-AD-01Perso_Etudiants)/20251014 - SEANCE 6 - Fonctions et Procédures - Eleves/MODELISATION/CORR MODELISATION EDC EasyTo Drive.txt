' Définition des classes (tables)
class AutoEcole {
  + id: int (PK)
  raisonSociale: varchar
  codePostal: varchar
  ville: varchar
}

class Utilisateur {
  + id: int (PK)
  nom: varchar
  prénom: varchar
}

class Role {
  + id: int (PK)
  nom: varchar
}

class Action {
  + id: int (PK)
  libelle: varchar
}

class TableDb {
  + id: int (PK)
  nom: varchar
}

' Evenement en tant que classe d'association
class Evenement {
  dateHeure: datetime
  idEnregistrement: int
}

' Définition des relations
Utilisateur "1..*" -- "1" AutoEcole : "Dépend de"
Utilisateur "   *" -- "1" Role : "Avoir"

' Déclaration de la classe d'association
(Utilisateur, Action) .. Evenement

' Un événement est lié à une table de la base de données
Evenement "0..*" -- "1" TableDb : "concerne la table"

@enduml
