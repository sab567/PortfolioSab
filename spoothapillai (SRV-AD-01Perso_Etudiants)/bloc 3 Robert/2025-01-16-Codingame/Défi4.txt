line = input()



a = int(line[0])
operator = line[1] 
b = int(line[2])  

if operator == '+':
    result = a + b
elif operator == '-':
    result = a - b
elif operator == '*':
    result = a * b
elif operator == '/':
    result = round(a / b) 

# Write an answer using print

print(result)
